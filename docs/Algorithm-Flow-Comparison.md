## SQL Query Comparison

### Original Algorithm - Nested Queries

```sql
-- Deeply nested, hard to read
SELECT col1, col2, ..., s.Depth
INTO #tmp
FROM (
    SELECT DISTINCT f.col1, f.col2, ..., s.Depth
    FROM TableA f
    INNER JOIN (
        SELECT Key0, Key1, Depth
        FROM Processing s
        WHERE s.Iteration IN (
            SELECT FoundIteration 
            FROM Operations 
            WHERE Status = 0
        )
    ) s ON f.Key1 = s.Key0
    WHERE f.col1 IS NOT NULL
      AND NOT EXISTS (
          SELECT * 
          FROM ProcessingB p 
          WHERE p.Key0 = f.col1
      )
) x

INSERT INTO ProcessingB SELECT * FROM #tmp
-- Separate operations table update
-- Drop temp table
GO
```

### Refactored Algorithm - CTE-Based

```sql
-- Clean, readable, optimizable
WITH SourceRecords AS (
    SELECT Key0, Key1, Depth, Fk
    FROM Processing
    WHERE Iteration IN (
        SELECT FoundIteration 
        FROM Operations 
        WHERE Status = 0 AND SessionId = 'xxx'
    )
),
NewRecords AS (
    SELECT DISTINCT
        tgt.col1 AS Key0,
        tgt.col2 AS Key1,
        src.Depth + 1 AS Depth
    FROM TableA tgt
    INNER JOIN SourceRecords src 
        ON tgt.Key1 = src.Key0
    WHERE tgt.col1 IS NOT NULL
      AND NOT EXISTS (
          SELECT 1 
          FROM ProcessingB existing 
          WHERE existing.Key0 = tgt.col1
            AND existing.Color = 1
      )
)
INSERT INTO ProcessingB (Key0, Key1, Color, Source, Depth, Fk, Iteration)
SELECT *, 1, 5, Depth, 42, 6
FROM NewRecords;

-- Operations update in same batch
INSERT INTO Operations (...)
SELECT ... FROM NewRecords GROUP BY Depth;
GO
```
